.demo-box {
    display: inline-block;
    min-width: 325pt;
    height: 60pt;
    padding-left: 10pt;
    
    /* 多背景图：左侧、中间、右侧，按从下到上的顺序叠加 */
    background-image: 
        url('./images/course-title-left.svg'),    /* 左侧装饰图 */
        url('./images/course-title-center.svg'),   /* 中间重复图 */
        url('./images/course-title-right.svg');    /* 右侧装饰图 */
    
    /* 背景图尺寸：分别对应三张图片 */
    background-size: 
        80pt 60pt,      /* 左侧图：固定尺寸 */
        auto 50pt,      /* 中间图：高度固定50pt，宽度auto保持宽高比，可横向拉伸 */
        33pt 50pt;      /* 右侧图：固定尺寸 */
    
    /* 背景图重复方式：分别对应三张图片 */
    background-repeat: 
        no-repeat,      /* 左侧图：不重复 */
        repeat-x,       /* 中间图：横向重复，实现自适应填充 */
        no-repeat;      /* 右侧图：不重复 */
    
    /* 背景图位置：分别对应三张图片 */
    background-position: 
        0 0,                                    /* 左侧图：左上角 */
        80pt 5pt,                              /* 中间图：距离左侧80pt，距离顶部5pt */
        100% calc(100% - 5pt);                 /* 右侧图：右侧对齐，距离底部5pt */
    
    /* 背景图裁剪区域：分别对应三张图片 */
    background-clip: 
        border-box,     /* 左侧图：延伸到边框区域 */
        content-box,    /* 中间图：仅在内容区域显示 */
        border-box;     /* 右侧图：延伸到边框区域 */
}
